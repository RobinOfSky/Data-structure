#### Node

public class Node {
    public int data;
    public  Node next;

    public  Node(){}
    public  Node(int data){
        this.data=data;
    }
}

#### CircularLinkedlist

```java
package com.robin.Circularlinkedlist;


public class CircularLinkedlist {
    private  Node head;
    private Node currect;

    public CircularLinkedlist(){};
    public  CircularLinkedlist(Node node){
        this.head=node;
    }

    /**
     * 插入结点
     * 1.头插法
     * 2.尾插法
     * 3指定位置插入
     */

    //插入头结点
    public  void insertFirst(int data){
        if(head==null){
            System.out.println("链表为空，将插入的结点作为头结点！");
            head.data= data;
        }else{
            currect=head;
            Node insertNode=new Node(data);
            insertNode.next=head;
            head.next=insertNode;
        }
    }

    //尾插法
    public  void insertTail(int data){
        if(head==null){
            System.out.println("链表为空，将插入的结点作为头结点！");
            head.data= data;
        }else{
            currect=head;
            while(currect.next!=null) {
                currect = currect.next;
            }
                Node insertNode=new Node(data);
                currect.next=insertNode;
                insertNode.next=head;

        }
    }

    //遍历
    public void display(){
        currect =head;
        while(currect!=null){
            System.out.println(currect.data);
            currect=currect.next;
        }
    }
}
```
